### 第 9 题
```
SELECT 
  ROW_NUMBER() OVER(PARTITION BY class_id ORDER BY score DESC) AS rank_num,
  student_name, class_id, score
FROM student_score;


SELECT 
  -- 核心：分组+排序+生成排名
  ROW_NUMBER() OVER(
    PARTITION BY 分组字段  -- 按什么维度分组（比如班级、部门）
    ORDER BY 排序字段 排序方式  -- 组内按什么排序（比如分数降序、业绩升序）
  ) AS 排名字段名,
  -- 以下是你要展示的业务字段
  字段1, 字段2, 字段3
FROM 数据表名;

### 第 10 题

WITH high_score_stu AS (
  SELECT student_id, student_name, score FROM student_score WHERE score >= 90
)
SELECT * FROM student s
JOIN high_score_stu h ON s.student_id = h.student_id;


#### 固定模板

-- 第一步：定义CTE（临时中间表），可以理解为“临时视图”
WITH 临时表别名 AS (
  -- 子查询：定义临时表的内容（筛选、聚合、关联都可以）
  SELECT 字段1, 字段2 FROM 数据表名 WHERE 筛选条件
)
-- 第二步：使用这个临时表做后续查询（关联、筛选、聚合等）
SELECT * FROM 主表 
JOIN 临时表别名 ON 主表.关联字段 = 临时表别名.关联字段;



```
