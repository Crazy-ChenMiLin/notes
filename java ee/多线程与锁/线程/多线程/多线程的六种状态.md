public class ThreadSixStatesDemo {
    // 全局锁对象，用于制造BLOCKED状态（竞争synchronized锁）
    private static final Object LOCK = new Object();

    public static void main(String[] args) throws InterruptedException {
        // ==================== 1. 演示 NEW 状态 ====================
        Thread newThread = new Thread(() -> {});
        System.out.println("1. 线程状态（NEW）：" + newThread.getState());

        // ==================== 2. 演示 RUNNABLE 状态 ====================
        Thread runnableThread = new Thread(() -> {
            // 循环执行一小段时间，让主线程有机会获取其RUNNABLE状态
            long startTime = System.currentTimeMillis();
            while (System.currentTimeMillis() - startTime < 500) {
                // 空循环，保持线程运行
            }
            System.out.println("   → runnableThread 执行中（RUNNABLE状态内运行）");
        });
        runnableThread.start();
        // 主线程稍等100ms，确保runnableThread已启动且处于RUNNABLE状态
        Thread.sleep(100);
        System.out.println("2. 线程状态（RUNNABLE）：" + runnableThread.getState());
        // 等待runnableThread执行完毕，避免影响后续演示
        runnableThread.join();

        // ==================== 3. 演示 BLOCKED 状态 ====================
        // 第一个线程：获取LOCK锁，然后休眠1秒（持有锁不释放）
        Thread holdLockThread = new Thread(() -> {
            synchronized (LOCK) {
                try {
                    // 休眠1秒，持有锁期间，让另一个线程等待锁
                    Thread.sleep(1000);
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
        });
        holdLockThread.start();

        // 第二个线程：竞争LOCK锁，此时会被阻塞（BLOCKED状态）
        Thread blockedThread = new Thread(() -> {
            synchronized (LOCK) {
                // 只有当holdLockThread释放锁后，才会执行到这里
                System.out.println("   → blockedThread 获取到锁，退出阻塞");
            }
        });
        blockedThread.start();
        // 主线程稍等100ms，确保blockedThread进入阻塞状态
        Thread.sleep(100);
        System.out.println("3. 线程状态（BLOCKED）：" + blockedThread.getState());
        // 等待两个线程执行完毕
        holdLockThread.join();
        blockedThread.join();

        // ==================== 4. 演示 WAITING 状态 ====================
        Thread waitingThread = new Thread(() -> {
            synchronized (LOCK) {
                try {
                    // 无超时等待，必须由其他线程调用 LOCK.notify() / notifyAll() 唤醒
                    LOCK.wait();
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
        });
        waitingThread.start();
        // 主线程稍等100ms，确保waitingThread进入WAITING状态
        Thread.sleep(100);
        System.out.println("4. 线程状态（WAITING）：" + waitingThread.getState());
        // 唤醒waitingThread（避免其一直等待）
        synchronized (LOCK) {
            LOCK.notify();
        }
        waitingThread.join();

        // ==================== 5. 演示 TIMED_WAITING 状态 ====================
        Thread timedWaitingThread = new Thread(() -> {
            try {
                // 有超时等待（200ms），超时后自动唤醒，无需其他线程干预
                Thread.sleep(200);
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        });
        timedWaitingThread.start();
        // 主线程稍等100ms，确保timedWaitingThread进入TIMED_WAITING状态
        Thread.sleep(100);
        System.out.println("5. 线程状态（TIMED_WAITING）：" + timedWaitingThread.getState());
        timedWaitingThread.join();

        // ==================== 6. 演示 TERMINATED 状态 ====================
        Thread terminatedThread = new Thread(() -> {
            System.out.println("   → terminatedThread 执行完毕，即将进入TERMINATED状态");
        });
        terminatedThread.start();
        // 等待线程执行完毕
        terminatedThread.join();
        System.out.println("6. 线程状态（TERMINATED）：" + terminatedThread.getState());
    }
}
