## 四、批量查询功能（21:50-22:30）

### 1. 新建 `step2.py`，复制以下代码


```
import pymongo
import requests
import time

# ----------------------
# 1. 初始化
# ----------------------
client = pymongo.MongoClient("mongodb://localhost:27017/")
db = client["word_db"]
collection = db["words"]

headers = {
    "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/122.0.0.0 Safari/537.36"
}

# ----------------------
# 2. 批量查询函数
# ----------------------
def batch_query(count=100):
    print(f"开始批量查询 {count} 条单词...")
    start_time = time.time()  # 记录开始时间
    
    success_count = 0
    # 从数据库取 count 条单词
    words = collection.find().limit(count)
    
    for idx, word_entry in enumerate(words):
        target_word = word_entry["word"]
        try:
            api_url = f"https://dict.youdao.com/suggest?num=1&ver=3.0&doctype=json&cache=false&le=en&q={target_word}"
            response = requests.get(api_url, headers=headers, timeout=3)
            
            if response.status_code == 200:
                success_count += 1
                # 每10条打印一次进度
                if (idx + 1) % 10 == 0:
                    print(f"已完成: {idx + 1}/{count}")
                    
        except Exception as e:
            print(f"查询出错: {target_word}, 错误: {e}")
    
    # 统计耗时
    end_time = time.time()
    total_time = end_time - start_time
    print("\n" + "="*30)
    print(f"批量查询完成！")
    print(f"成功查询: {success_count} 条")
    print(f"总耗时: {total_time:.2f} 秒")
    print(f"平均每条: {total_time/count:.2f} 秒")
    print("="*30)

# ----------------------
# 3. 执行批量查询
# ----------------------
if __name__ == "__main__":
    batch_query(100)
```

同样在终端输入：

```

## 五、异常处理与日志（22:30-23:00）

### 1. 新建 `step3.py`，复制最终版代码

python

运行

```
import pymongo
import requests
import time
import random
from datetime import datetime

# ----------------------
# 1. 日志输出函数（带时间戳）
# ----------------------
def log(message):
    print(f"[{datetime.now().strftime('%Y-%m-%d %H:%M:%S')}] {message}")

# ----------------------
# 2. 带重试的查询函数
# ----------------------
def query_with_retry(word, headers, max_retries=3):
    for attempt in range(max_retries):
        try:
            # 随机等待 0.1-0.5 秒，防止请求过快被封
            time.sleep(random.uniform(0.1, 0.5))
            
            api_url = f"https://dict.youdao.com/suggest?num=1&ver=3.0&doctype=json&cache=false&le=en&q={word}"
            response = requests.get(api_url, headers=headers, timeout=5)
            
            if response.status_code == 200:
                return True, "查询成功"
            else:
                log(f"警告: {word} 返回状态码 {response.status_code} (尝试 {attempt+1}/{max_retries})")
                
        except Exception as e:
            log(f"错误: {word} 发生异常 {str(e)} (尝试 {attempt+1}/{max_retries})")
    
    return False, "重试失败"

# ----------------------
# 3. 主程序
# ----------------------
def main():
    log("程序启动...")
    
    # 连接数据库
    try:
        client = pymongo.MongoClient("mongodb://localhost:27017/", serverSelectionTimeoutMS=5000)
        client.server_info()  # 测试连接
        db = client["word_db"]
        collection = db["words"]
        log("数据库连接成功")
    except Exception as e:
        log(f"数据库连接失败: {e}")
        return

    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/122.0.0.0 Safari/537.36"
    }

    # 执行批量查询
    count = 100
    log(f"开始批量查询 {count} 条单词")
    start_time = time.time()
    
    success = 0
    words = list(collection.find().limit(count))  # 转为列表方便计数
    
    for idx, word_entry in enumerate(words):
        word = word_entry["word"]
        is_success, msg = query_with_retry(word, headers)
        
        if is_success:
            success += 1
        
        # 进度日志
        if (idx + 1) % 20 == 0:
            log(f"进度: {idx + 1}/{count} | 成功: {success}")
    
    # 最终统计
    total_time = time.time() - start_time
    log("\n" + "="*40)
    log(f"任务完成！")
    log(f"总耗时: {total_time:.2f}s | 成功: {success}/{count}")
    log(f"成功率: {success/count*100:.1f}%")
    log("="*40)

if __name__ == "__main__":
    main()
```

运行

```

现在你会看到带时间戳的日志，失败的请求会自动重试 3 次，还加了随机延迟防止被接口封禁。
